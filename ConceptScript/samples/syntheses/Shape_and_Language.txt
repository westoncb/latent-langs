        IMPLEMENT FUNC CompileSketch(w : Language(Lexicon.Symbols)) -> SemanticSketch {
          syn := SyntaxTree(DerivationTree(w, SyntaxRules))

          RETURN sem

        IMPLEMENT FUNC CompileShape(sk : SemanticSketch) -> GeometricForm {
          objs := { (o, Embedding(o)) | (s, o, Embedding) ‚àà ObjectSchemas, o ‚àà sk.Objects }
          rels := { (a, b) ‚Ü¶ Constraint | ((a, r, b), Constraint) ‚àà sk.Relations }

            DisjointUnion({ o | (o, _) ‚àà objs }),
            { Embedding(a)(p) = Embedding(b)(q) | ((a, b), Constraint) ‚àà rels, Constraint(p, q) }  
          )
        }
      }

          syn := SyntaxTree(
            Root := SyntaxNode(Phrase := "start"),
            Nodes := { SyntaxNode(Phrase := ‚ü¶o‚üß) | o ‚àà sem.Objects } ‚à™ 
                     { SyntaxNode(Phrase := s) | (a, (s, r, c), b) ‚àà sem.Relations },
            Edges := { (SyntaxNode(‚ü¶a‚üß), SyntaxNode(s)) | (a, (s, r, c), b) ‚àà sem.Relations } ‚à™
